\documentclass{standalone}
\usepackage{tikz}

\begin{document}

%x description="draw a line using shift"
\begin{tikzpicture}
	\draw[gray] (0,0) circle (0.1);
%x step={
	\draw (0,0) -- (1,1) 
		% use (1,0.5) as new origin
		[shift= {(1,0.5)} ]  % {} are necessary
		(0,0) -- (1,1);

	% or
	\draw (0,0) -- (1,1) 
		[xshift=1cm, yshift=0.5cm]
		(0,0) -- (1,1);
%x }
\end{tikzpicture}


\end{document}
